    ```python
    """BMXF field-weighted retriever implementing the ToolRetriever interface.

    Adapts BMXIndex to score ToolMapping objects using field-weighted BMX scoring
    across 5 fields: tool_name (3.0), namespace (2.5), retrieval_aliases (1.5),
    description (1.0), parameter_names (0.5).

    Follows KeywordRetriever.rebuild_index(registry) lifecycle pattern.
    Tool keys use server_name__tool_name format (double underscore via _make_key()).
    """

    from __future__ import annotations

    import re
    from typing import TYPE_CHECKING

    from .base import ToolRetriever
    from .bmx_index import BMXIndex
    from .models import RetrievalConfig, RetrievalContext, ScoredTool, ToolDoc
    from .namespace_filter import compute_namespace_boosts

    if TYPE_CHECKING:
        from src.multimcp.mcp_proxy import ToolMapping


    # ── Curated alias maps ──────────────────────────────────────────────────

    NAMESPACE_ALIASES: dict[str, list[str]] = {
        "github": ["repository", "pull_request", "issue", "git", "code_review", "branch", "commit"],
        "brave-search": ["web_search", "internet", "lookup", "find", "query"],
        "context7": ["documentation", "library", "docs", "api_reference", "examples"],
        "docker": ["container", "image", "compose", "deploy", "service"],
        "filesystem": ["file", "directory", "read", "write", "path", "folder"],
        "shell": ["terminal", "command", "bash", "exec", "run", "process"],
        "slack": ["message", "channel", "chat", "notification"],
        "npm": ["package", "install", "node", "dependency"],
        "pip": ["package", "install", "python", "dependency"],
        "cargo": ["crate", "build", "rust", "compile"],
        "kubectl": ["kubernetes", "pod", "deployment", "service", "cluster"],
        "terraform": ["infrastructure", "cloud", "provision", "iac"],
    }

    ACTION_ALIASES: dict[str, list[str]] = {
        "list": ["get", "fetch", "show", "enumerate"],
        "create": ["add", "new", "make", "insert"],
        "search": ["find", "query", "lookup"],
        "delete": ["remove", "destroy", "drop"],
        "update": ["edit", "modify", "change", "patch"],
        "run": ["execute", "invoke", "start"],
        "get": ["fetch", "read", "retrieve"],
    }


    # ── Helpers ─────────────────────────────────────────────────────────────

    def _extract_param_names(input_schema: object) -> list[str]:
        """Extract property keys from a tool's inputSchema."""
        if not isinstance(input_schema, dict):
            return []
        props = input_schema.get("properties", {})
        if not isinstance(props, dict):
            return []
        return list(props.keys())


    def _generate_aliases(namespace: str, tool_name: str) -> list[str]:
        """Generate curated lexical aliases for retrieval vocabulary bridging.

        Combines namespace-level aliases (from NAMESPACE_ALIASES) with action-verb
        synonyms derived from the tool name (from ACTION_ALIASES).

        A tool named "search_repositories" in namespace "github" receives:
        - Namespace aliases: repository, pull_request, issue, git, ...
        - Action aliases (for "search"): find, query, lookup
        The name tokens themselves ("search", "repositories") are already indexed
        in the tool_name field — aliases add vocabulary, not duplicates.
        """
        aliases: list[str] = []

        # Namespace aliases
        ns_aliases = NAMESPACE_ALIASES.get(namespace, [])
        aliases.extend(ns_aliases)

        # Action aliases from tool name tokens
        name_tokens = re.split(r"[_\W]+", tool_name.lower())
        for token in name_tokens:
            action_syns = ACTION_ALIASES.get(token, [])
            aliases.extend(action_syns)

        # Deduplicate while preserving order
        seen: set[str] = set()
        unique: list[str] = []
        for a in aliases:
            if a not in seen:
                seen.add(a)
                unique.append(a)
        return unique


    # ── BMXFRetriever ────────────────────────────────────────────────────────

    class BMXFRetriever(ToolRetriever):
        """BMXF field-weighted retriever.

        Implements ToolRetriever ABC. Follows KeywordRetriever.rebuild_index()
        lifecycle: call rebuild_index(registry) after any tool_to_server change,
        then call retrieve() per-request.

        BMX parameters (alpha, beta) are auto-tuned from corpus statistics —
        no manual k1/b tuning required.
        """

        def __init__(self, config: RetrievalConfig) -> None:
            self._config = config
            self._index = BMXIndex()     # alpha/beta auto-tuned from corpus
            self._tool_docs: dict[str, ToolDoc] = {}

        # ── Index lifecycle ──────────────────────────────────────────────

        def rebuild_index(self, registry: "dict[str, ToolMapping]") -> None:
            """Rebuild BMXF field index from tool registry.

            Mirrors KeywordRetriever.rebuild_index() — call after any
            tool_to_server mutation (register_client / unregister_client).

            Tool keys use server_name__tool_name format (double underscore,
            matching MCPProxyServer._make_key()).
            """
            self._tool_docs.clear()

            for key, mapping in registry.items():
                ns, name = key.split("__", 1) if "__" in key else ("", key)
                param_names = _extract_param_names(mapping.tool.inputSchema)
                aliases = _generate_aliases(ns, name)
                self._tool_docs[key] = ToolDoc(
                    tool_key=key,
                    tool_name=name,
                    namespace=ns,
                    description=mapping.tool.description or "",
                    parameter_names=" ".join(param_names),
                    retrieval_aliases=" ".join(aliases),
                )

            self._index.build_field_index(list(self._tool_docs.values()))

        # ── Retrieval ────────────────────────────────────────────────────

        async def retrieve(
            self,
            context: RetrievalContext,
            candidates: list["ToolMapping"],
        ) -> list[ScoredTool]:
            """Score candidates against context query using BMXF.

            Empty query → all candidates returned with uniform score=0.5,
            limited to top_k. This matches KeywordRetriever's empty-query behavior.

            Non-empty query → BMXF field-weighted search, namespace boosts applied,
            results sorted descending by score.
            """
            # Build candidate lookup: tool_key -> ToolMapping
            candidate_keys: dict[str, "ToolMapping"] = {
                f"{m.server_name}__{m.tool.name}": m for m in candidates
            }

            # Empty query: uniform score
            if not context.query:
                scored = [
                    ScoredTool(
                        tool_key=key,
                        tool_mapping=mapping,
                        score=0.5,
                        tier="full",
                    )
                    for key, mapping in candidate_keys.items()
                ]
                return scored[:self._config.top_k]

            # BMXF search: top_k*2 to allow namespace boost reordering
            results = self._index.search_fields(context.query, top_k=self._config.top_k * 2)

            # Namespace boosts (same as KeywordRetriever)
            boosts = compute_namespace_boosts(
                {k: v for k, v in candidate_keys.items()},
                server_hint=context.server_hint,
            )

            scored: list[ScoredTool] = []
            for chunk_id, score in results:
                if chunk_id not in candidate_keys:
                    continue
                boost = boosts.get(chunk_id, 1.0)
                scored.append(ScoredTool(
                    tool_key=chunk_id,
                    tool_mapping=candidate_keys[chunk_id],
                    score=score * boost,
                    tier="full",
                ))

            scored.sort(key=lambda s: s.score, reverse=True)
            return scored[:self._config.top_k]
    ```

    ```python
    """Tests for ToolCatalogSnapshot creation and schema_hash stability."""
    import pytest
    from unittest.mock import MagicMock
    from mcp import types

    def _make_mapping(server: str, tool_name: str, desc: str, props: dict | None = None):

        def _build_registry(self):
            return {
                "github__search": _make_mapping(
                    "github", "search", "Search GitHub repositories",
                    {"query": {}, "owner": {}}
                ),
                "exa__web_search": _make_mapping(
                    "exa", "web_search", "Search the web"
                ),
            }

        def test_empty_registry_returns_valid_snapshot(self):
            snap = build_snapshot({})
            assert snap.docs == []
            assert len(snap.schema_hash) == 64
            assert snap.version != ""

        def test_snapshot_has_one_doc_per_tool(self):
            snap = build_snapshot(self._build_registry())
            assert len(snap.docs) == 2

        def test_tool_doc_fields_correct(self):
            snap = build_snapshot(self._build_registry())
            doc = next(d for d in snap.docs if d.tool_key == "github__search")
            assert doc.tool_name == "search"
            assert doc.namespace == "github"
            assert doc.description == "Search GitHub repositories"
            assert "query" in doc.parameter_names
            assert doc.retrieval_aliases == ""  # Not populated by build_snapshot

        def test_schema_hash_is_stable_for_identical_registry(self):
            registry = self._build_registry()
            snap1 = build_snapshot(registry)
            snap2 = build_snapshot(registry)
            assert snap1.schema_hash == snap2.schema_hash

        def test_schema_hash_changes_on_description_change(self):
            registry1 = self._build_registry()
            registry2 = {
                "github__search": _make_mapping("github", "search", "CHANGED DESCRIPTION"),
                "exa__web_search": _make_mapping("exa", "web_search", "Search the web"),
            }
            snap1 = build_snapshot(registry1)
            snap2 = build_snapshot(registry2)
            assert snap1.schema_hash != snap2.schema_hash

        def test_schema_hash_changes_on_new_tool(self):
            registry1 = self._build_registry()
            registry2 = dict(self._build_registry())
            registry2["new__tool"] = _make_mapping("new", "tool", "A new tool")
            snap1 = build_snapshot(registry1)
            snap2 = build_snapshot(registry2)
            assert snap1.schema_hash != snap2.schema_hash

        def test_version_increments_each_call(self):
            snap1 = build_snapshot({})
            snap2 = build_snapshot({})
            assert int(snap2.version) == int(snap1.version) + 1

        def test_built_at_is_recent_timestamp(self):
            import time
            before = time.time()
            snap = build_snapshot({})
            after = time.time()
            assert before <= snap.built_at <= after

        def test_schema_hash_is_64_char_hex(self):
            snap = build_snapshot(self._build_registry())
            assert len(snap.schema_hash) == 64
            assert all(c in "0123456789abcdef" for c in snap.schema_hash)
    ```
