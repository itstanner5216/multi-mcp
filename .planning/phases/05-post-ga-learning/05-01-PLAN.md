From RankingEvent JSONL (produced by FileRetrievalLogger):
  Each line has: direct_tool_calls: list[str], active_tool_ids: list[str], session_id: str

PPMI formula:
  PMI(x,y) = log2(P(x,y) / (P(x) * P(y)))
  PPMI(x,y) = max(0, PMI(x,y))
  P(x,y) = count(x,y used in same session) / total_sessions
  P(x) = count(sessions using x) / total_sessions

  base_tokens: dict[str, float] — e.g. {"manifest:Cargo.toml": 3.0, "lang:rust": 2.0}
  ppmi_weights: dict[tuple[str,str], float] — e.g. {("cargo__build", "cargo__test"): 2.5}

"""Co-occurrence tracking and PPMI computation for usage-based learning.

Builds a co-occurrence graph from production RankingEvent logs, computes
Positive Pointwise Mutual Information (PPMI) for tool pairs, and provides
a token reweighting function that incorporates learned associations.

Requires: >50 sessions of collected data for meaningful statistics.
"""

from __future__ import annotations

import json
import math
from collections import Counter, defaultdict
from dataclasses import dataclass, field
from pathlib import Path


@dataclass
class CooccurrenceTracker:
    """Tracks tool co-occurrence within sessions.

    Records which tools appear together in the same session to build
    a co-occurrence matrix for PPMI computation.
    """

    pair_counts: Counter = field(default_factory=Counter)     # (tool_a, tool_b) -> count
    tool_counts: Counter = field(default_factory=Counter)     # tool -> session count
    total_sessions: int = 0

    def record_session(self, tools_used: list[str]) -> None:
        """Record tools used in a single session.

        Args:
            tools_used: List of tool keys used in this session (may have duplicates).
        """
        unique_tools = sorted(set(tools_used))
        if not unique_tools:
            return

        self.total_sessions += 1

        for tool in unique_tools:
            self.tool_counts[tool] += 1

        # Record all unique pairs (ordered to avoid (a,b) and (b,a) duplication)
        for i in range(len(unique_tools)):
            for j in range(i + 1, len(unique_tools)):
                pair = (unique_tools[i], unique_tools[j])
                self.pair_counts[pair] += 1

    @classmethod
    def from_jsonl(cls, log_path: str | Path) -> "CooccurrenceTracker":
        """Build tracker from RankingEvent JSONL log file.

        Groups events by session_id and collects direct_tool_calls per session.
        """
        tracker = cls()
        path = Path(log_path)
        if not path.exists():
            return tracker

        session_tools: dict[str, set[str]] = defaultdict(set)

        with open(path, "r", encoding="utf-8") as f:
            for line in f:
                line = line.strip()
                if not line:
                    continue
                try:
                    event = json.loads(line)
                except json.JSONDecodeError:
                    continue

                # Skip non-ranking-event lines (e.g. alert lines)
                if event.get("type") == "alert":
                    continue

                session_id = event.get("session_id", "")
                calls = event.get("direct_tool_calls", [])
                if calls:
                    session_tools[session_id].update(calls)

        for tools in session_tools.values():
            tracker.record_session(list(tools))

        return tracker


def compute_ppmi(
    tracker: CooccurrenceTracker,
    min_count: int = 2,
) -> dict[tuple[str, str], float]:
    """Compute Positive Pointwise Mutual Information for all tool pairs.

    PMI(x,y) = log2(P(x,y) / (P(x) * P(y)))
    PPMI(x,y) = max(0, PMI(x,y))

    Args:
        tracker: CooccurrenceTracker with recorded sessions.
        min_count: Minimum pair count to include (filters noise from rare pairs).

    Returns:
        Dict of (tool_a, tool_b) -> PPMI score. Only pairs with PPMI > 0 included.
    """
    if tracker.total_sessions == 0:
        return {}

    n = tracker.total_sessions
    ppmi_scores: dict[tuple[str, str], float] = {}

    for (a, b), count_ab in tracker.pair_counts.items():
        if count_ab < min_count:
            continue

        count_a = tracker.tool_counts.get(a, 0)
        count_b = tracker.tool_counts.get(b, 0)

        if count_a == 0 or count_b == 0:
            continue

        p_ab = count_ab / n
        p_a = count_a / n
        p_b = count_b / n

        pmi = math.log2(p_ab / (p_a * p_b))
        ppmi = max(0.0, pmi)

        if ppmi > 0:
            ppmi_scores[(a, b)] = ppmi

    return ppmi_scores


def reweight_tokens(
    base_tokens: dict[str, float],
    ppmi_scores: dict[tuple[str, str], float],
    active_tools: list[str],
    boost_factor: float = 0.3,
) -> dict[str, float]:
    """Reweight environment tokens using PPMI associations.

    For each active tool, find its PPMI associations with other tools.
    Boost tokens associated with tools that co-occur with active tools.

    This is a soft signal: PPMI adjusts token weights by boost_factor * ppmi_score,
    not a hard override.

    Args:
        base_tokens: Original token weights (e.g. from telemetry).
        ppmi_scores: PPMI scores from compute_ppmi().
        active_tools: Currently active tools in the session.
        boost_factor: Multiplier for PPMI boost (default 0.3 — conservative).

    Returns:
        New token dict with adjusted weights.
    """
    adjusted = dict(base_tokens)

    # Build a lookup: tool -> list of (partner_tool, ppmi_score)
    tool_associations: dict[str, list[tuple[str, float]]] = defaultdict(list)
    for (a, b), score in ppmi_scores.items():
        tool_associations[a].append((b, score))
        tool_associations[b].append((a, score))

    # For each active tool, boost tokens of associated tools
    for tool in active_tools:
        for partner, ppmi in tool_associations.get(tool, []):
            # Extract namespace from partner tool key
            ns = partner.split("__", 1)[0] if "__" in partner else partner
            token_key = f"cooccur:{ns}"
            current = adjusted.get(token_key, 0.0)
            adjusted[token_key] = current + boost_factor * ppmi

    return adjusted

"""Tests for co-occurrence tracking and PPMI computation (Phase 5)."""

from __future__ import annotations

import dataclasses
import json
import math
from pathlib import Path

import pytest

from src.multimcp.retrieval.learning import (
    CooccurrenceTracker,
    compute_ppmi,
    reweight_tokens,
)
from src.multimcp.retrieval.models import RankingEvent


class TestCooccurrenceTracker:
    def test_empty_tracker(self) -> None:
        t = CooccurrenceTracker()
        assert t.total_sessions == 0
        assert len(t.pair_counts) == 0

    def test_single_session_pair(self) -> None:
        t = CooccurrenceTracker()
        t.record_session(["a", "b"])
        assert t.total_sessions == 1
        assert t.pair_counts[("a", "b")] == 1
        assert t.tool_counts["a"] == 1
        assert t.tool_counts["b"] == 1

    def test_single_tool_no_pairs(self) -> None:
        t = CooccurrenceTracker()
        t.record_session(["a"])
        assert t.total_sessions == 1
        assert t.tool_counts["a"] == 1
        assert len(t.pair_counts) == 0

    def test_empty_session_ignored(self) -> None:
        t = CooccurrenceTracker()
        t.record_session([])
        assert t.total_sessions == 0

    def test_multiple_sessions(self) -> None:
        t = CooccurrenceTracker()
        t.record_session(["a", "b"])
        t.record_session(["a", "c"])
        assert t.total_sessions == 2
        assert t.tool_counts["a"] == 2
        assert t.tool_counts["b"] == 1
        assert t.tool_counts["c"] == 1
        assert t.pair_counts[("a", "b")] == 1
        assert t.pair_counts[("a", "c")] == 1

    def test_duplicates_deduplicated(self) -> None:
        t = CooccurrenceTracker()
        t.record_session(["a", "b", "a", "b"])
        assert t.tool_counts["a"] == 1
        assert t.pair_counts[("a", "b")] == 1

    def test_three_tools_three_pairs(self) -> None:
        t = CooccurrenceTracker()
        t.record_session(["a", "b", "c"])
        assert len(t.pair_counts) == 3
        assert t.pair_counts[("a", "b")] == 1
        assert t.pair_counts[("a", "c")] == 1
        assert t.pair_counts[("b", "c")] == 1

    def test_pairs_always_sorted(self) -> None:
        t = CooccurrenceTracker()
        t.record_session(["z", "a", "m"])
        # Pairs should be sorted: (a,m), (a,z), (m,z)
        assert ("a", "m") in t.pair_counts
        assert ("a", "z") in t.pair_counts
        assert ("m", "z") in t.pair_counts


class TestFromJsonl:
    def test_reads_jsonl(self, tmp_path: Path) -> None:
        log = tmp_path / "events.jsonl"
        events = [
            RankingEvent(session_id="s1", turn_number=0, catalog_version="v1",
                         direct_tool_calls=["github__search", "github__create_pr"]),

class TestComputePPMI:
    def test_empty_tracker(self) -> None:
        t = CooccurrenceTracker()
        assert compute_ppmi(t) == {}

    def test_always_together_high_ppmi(self) -> None:
        t = CooccurrenceTracker()
        for _ in range(10):
            t.record_session(["a", "b"])
        ppmi = compute_ppmi(t, min_count=2)
        # a and b always co-occur: P(a,b) = P(a) = P(b) = 1.0
        # PMI = log2(1.0 / (1.0 * 1.0)) = 0
        # With some other sessions to create differentiation:
        assert ("a", "b") not in ppmi or ppmi[("a", "b")] == pytest.approx(0.0)

    def test_strong_association(self) -> None:
        t = CooccurrenceTracker()
        # a and b always together; c appears alone
        for _ in range(10):
            t.record_session(["a", "b"])
        for _ in range(10):
            t.record_session(["c"])
        ppmi = compute_ppmi(t, min_count=2)
        # P(a,b) = 10/20, P(a) = 10/20, P(b) = 10/20
        # PMI = log2(0.5 / (0.5 * 0.5)) = log2(2) = 1.0
        assert ppmi.get(("a", "b"), 0.0) == pytest.approx(1.0, abs=0.01)

    def test_min_count_filters(self) -> None:
        t = CooccurrenceTracker()
        t.record_session(["a", "b"])  # only 1 occurrence
        for _ in range(5):
            t.record_session(["c", "d"])
        ppmi = compute_ppmi(t, min_count=2)
        assert ("a", "b") not in ppmi
        # c,d should be present (5 occurrences)

    def test_ppmi_non_negative(self) -> None:
        t = CooccurrenceTracker()
        # Create independent tools: a appears in 90% sessions, b in 90%, but together only 50%
        for _ in range(50):
            t.record_session(["a", "b"])
        for _ in range(40):
            t.record_session(["a"])
        for _ in range(40):
            t.record_session(["b"])
        ppmi = compute_ppmi(t, min_count=2)
        for score in ppmi.values():
            assert score >= 0.0


class TestReweightTokens:
    def test_no_ppmi_no_change(self) -> None:
        base = {"lang:python": 2.0, "manifest:pyproject.toml": 3.0}
        result = reweight_tokens(base, {}, [])
        assert result == base

    def test_ppmi_boost_adds_token(self) -> None:
        base = {"lang:python": 2.0}
        ppmi = {("github__search", "github__create_pr"): 1.5}
        result = reweight_tokens(base, ppmi, ["github__search"], boost_factor=0.3)
        # Should add cooccur:github token
        assert "cooccur:github" in result
        assert result["cooccur:github"] > 0

    def test_boost_factor_controls_magnitude(self) -> None:
        base = {"lang:python": 2.0}
        ppmi = {("test__a", "test__b"): 2.0}
        r1 = reweight_tokens(base, ppmi, ["test__a"], boost_factor=0.1)
        r2 = reweight_tokens(base, ppmi, ["test__a"], boost_factor=0.5)
        assert r2.get("cooccur:test", 0) > r1.get("cooccur:test", 0)

    def test_base_tokens_preserved(self) -> None:
        base = {"lang:python": 2.0, "manifest:pyproject.toml": 3.0}
        ppmi = {("test__a", "test__b"): 1.0}
        result = reweight_tokens(base, ppmi, ["test__a"])
        assert result["lang:python"] == 2.0
        assert result["manifest:pyproject.toml"] == 3.0
