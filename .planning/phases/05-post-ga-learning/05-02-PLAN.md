<objective>
Create an exploration injection module that adds a small number of rarely-used tools to the active set. This prevents the BMXF scorer from creating a self-reinforcing feedback loop where popular tools stay popular and niche tools are never discovered.

Purpose: Without exploration, tools that are initially ranked low by BMXF will never be surfaced, so the system can't learn whether they should rank higher. By injecting 2 rarely-used tools per session, the system gains usage signal for the full tool catalog over time.

Output: src/multimcp/retrieval/exploration.py, tests/test_exploration.py
</objective>

```python
"""Exploration injection for active set diversity.

Adds a small number of rarely-used tools to the active set to prevent
feedback loops where only popular tools are surfaced. Behind feature flag.

Usage: Called by pipeline.get_tools_for_list() after BMXF scoring.
"""

from __future__ import annotations

import random
from collections import Counter
from dataclasses import dataclass, field


@dataclass
class ExplorationConfig:
    """Configuration for exploration injection."""

    enable_exploration: bool = False   # Feature flag (off by default)
    n_slots: int = 2                   # Number of exploration slots per session
    seed: int | None = None            # Random seed for reproducibility (testing)


class ExplorationInjector:
    """Selects rarely-used tools for injection into the active set.

    Selection is weighted by inverse frequency: tools used in fewer sessions
    have higher probability of being selected. This ensures the system
    explores its full tool catalog over time.
    """

    def __init__(self, config: ExplorationConfig | None = None) -> None:
        self._config = config or ExplorationConfig()
        self._rng = random.Random(self._config.seed)
        self._tool_usage: Counter = Counter()  # tool_key -> session count

    def update_usage(self, tool_counts: dict[str, int] | Counter) -> None:
        """Update tool usage frequency data.

        Args:
            tool_counts: Mapping of tool_key -> number of sessions where tool was used.
        """
        self._tool_usage = Counter(tool_counts)

    def inject(
        self,
        active_tools: set[str],
        all_tools: list[str],
        n_slots: int | None = None,
    ) -> list[str]:
        """Select tools for exploration injection.

        Args:
            active_tools: Tool keys already in the active set (will not be selected).
            all_tools: All available tool keys in the registry.
            n_slots: Override number of slots (default from config).

        Returns:
            List of tool keys to inject (at most n_slots, may be fewer if
            no eligible tools exist).
        """
        slots = n_slots if n_slots is not None else self._config.n_slots

        # Eligible: tools not already active
        eligible = [t for t in all_tools if t not in active_tools]
        if not eligible or slots <= 0:
            return []

        if not self._tool_usage:
            # No usage data: uniform random
            return self._rng.sample(eligible, min(slots, len(eligible)))

        # Inverse-frequency weighting: 1 / (usage_count + 1)
        # Tools with 0 usage get weight 1.0; tools with 10 usage get weight ~0.09
        weights = [1.0 / (self._tool_usage.get(t, 0) + 1) for t in eligible]

        # Weighted sampling without replacement
        selected: list[str] = []
        remaining = list(zip(eligible, weights))
        for _ in range(min(slots, len(remaining))):
            if not remaining:
                break
            tools, w = zip(*remaining)
            tools_list = list(tools)
            w_list = list(w)
            choice = self._rng.choices(tools_list, weights=w_list, k=1)[0]
            selected.append(choice)
            remaining = [(t, wt) for t, wt in remaining if t != choice]

        return selected
```

```python
"""Tests for exploration injection (Phase 5)."""

from __future__ import annotations

from collections import Counter

import pytest

from src.multimcp.retrieval.exploration import ExplorationConfig, ExplorationInjector


class TestExplorationInjector:
    def test_all_active_returns_empty(self) -> None:
        inj = ExplorationInjector(ExplorationConfig(seed=42))
        result = inj.inject({"a", "b", "c"}, ["a", "b", "c"])
        assert result == []

    def test_zero_slots_returns_empty(self) -> None:
        inj = ExplorationInjector(ExplorationConfig(seed=42, n_slots=0))
        result = inj.inject({"a"}, ["a", "b", "c"])
        assert result == []

    def test_respects_n_slots(self) -> None:
        inj = ExplorationInjector(ExplorationConfig(seed=42, n_slots=2))
        result = inj.inject(set(), ["a", "b", "c", "d", "e"])
        assert len(result) == 2

    def test_injected_not_in_active(self) -> None:
        inj = ExplorationInjector(ExplorationConfig(seed=42, n_slots=3))
        active = {"a", "b"}
        result = inj.inject(active, ["a", "b", "c", "d", "e"])
        for tool in result:
            assert tool not in active

    def test_deterministic_with_seed(self) -> None:
        config = ExplorationConfig(seed=123, n_slots=2)
        inj1 = ExplorationInjector(config)
        inj2 = ExplorationInjector(ExplorationConfig(seed=123, n_slots=2))
        all_tools = ["a", "b", "c", "d", "e"]
        r1 = inj1.inject(set(), all_tools)
        r2 = inj2.inject(set(), all_tools)
        assert r1 == r2

    def test_fewer_eligible_than_slots(self) -> None:
        inj = ExplorationInjector(ExplorationConfig(seed=42, n_slots=5))
        result = inj.inject({"a", "b"}, ["a", "b", "c"])
        assert len(result) == 1
        assert result[0] == "c"

    def test_empty_all_tools(self) -> None:
        inj = ExplorationInjector(ExplorationConfig(seed=42))
        result = inj.inject(set(), [])
        assert result == []

    def test_usage_weighting_prefers_rare(self) -> None:
        config = ExplorationConfig(seed=42, n_slots=1)
        inj = ExplorationInjector(config)
        inj.update_usage({"common": 100, "rare": 1, "never": 0})
        # Run many trials to check distribution
        counts: Counter = Counter()
        for seed in range(200):
            inj._rng.seed(seed)
            result = inj.inject(set(), ["common", "rare", "never"])
            if result:
                counts[result[0]] += 1
        # "never" should be picked most often, "common" least
        assert counts["never"] > counts["common"]

    def test_no_duplicates_in_result(self) -> None:
        inj = ExplorationInjector(ExplorationConfig(seed=42, n_slots=3))
        result = inj.inject(set(), ["a", "b", "c", "d"])
        assert len(result) == len(set(result))

    def test_override_n_slots(self) -> None:
        inj = ExplorationInjector(ExplorationConfig(seed=42, n_slots=2))
        result = inj.inject(set(), ["a", "b", "c", "d", "e"], n_slots=4)
        assert len(result) == 4
```
