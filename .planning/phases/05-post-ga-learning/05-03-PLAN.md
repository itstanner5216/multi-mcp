"""Integration tests for PPMI reweighting and exploration injection (Phase 5)."""

from __future__ import annotations

from unittest.mock import MagicMock

import pytest

from src.multimcp.retrieval.base import PassthroughRetriever
from src.multimcp.retrieval.bmx_retriever import BMXFRetriever
from src.multimcp.retrieval.logging import NullLogger
from src.multimcp.retrieval.models import RankingEvent, RetrievalConfig
from src.multimcp.retrieval.pipeline import RetrievalPipeline
from src.multimcp.retrieval.session import SessionStateManager


class _CapturingLogger(NullLogger):
    def __init__(self) -> None:
        self.events: list[RankingEvent] = []

    async def log_ranking_event(self, event: RankingEvent) -> None:
        self.events.append(event)


def _make_tool(name: str, server: str = "test") -> MagicMock:
    tool = MagicMock()
    tool.name = name
    tool.description = f"Test tool {name}"
    tool.inputSchema = {"type": "object", "properties": {}}
    mapping = MagicMock()
    mapping.tool = tool
    mapping.server_name = server
    return mapping


def _build_registry(n: int = 25) -> dict[str, MagicMock]:
    registry: dict[str, MagicMock] = {}
    for i in range(n):
        key = f"test__tool_{i:02d}"
        registry[key] = _make_tool(f"tool_{i:02d}")
    return registry


class TestLoadPPMI:
    def test_load_empty(self) -> None:
        config = RetrievalConfig()
        retriever = BMXFRetriever(config)
        retriever.load_ppmi({})
        assert retriever._ppmi_scores == {}

    def test_load_scores(self) -> None:
        config = RetrievalConfig()
        retriever = BMXFRetriever(config)
        scores = {("a", "b"): 1.5, ("c", "d"): 0.8}
        retriever.load_ppmi(scores)
        assert retriever._ppmi_scores == scores

    def test_load_replaces_existing(self) -> None:
        config = RetrievalConfig()
        retriever = BMXFRetriever(config)
        retriever.load_ppmi({("a", "b"): 1.0})
        retriever.load_ppmi({("c", "d"): 2.0})
        assert ("a", "b") not in retriever._ppmi_scores
        assert retriever._ppmi_scores[("c", "d")] == 2.0


class TestExplorationIntegration:
    @pytest.mark.asyncio
    async def test_exploration_disabled_no_injection(self) -> None:
        registry = _build_registry(25)
        logger = _CapturingLogger()
        config = RetrievalConfig(
            enabled=True, rollout_stage="ga",
            enable_exploration=False, max_k=15,
        )
        pipe = RetrievalPipeline(
            retriever=PassthroughRetriever(),
            session_manager=SessionStateManager(config),
            logger=logger,
            config=config,
            tool_registry=registry,
        )
        await pipe.get_tools_for_list("session-1")
        assert logger.events[0].exploration_tools == []

    @pytest.mark.asyncio
    async def test_exploration_enabled_injects_for_ga(self) -> None:
        registry = _build_registry(25)
        logger = _CapturingLogger()
        config = RetrievalConfig(
            enabled=True, rollout_stage="ga",
            enable_exploration=True, exploration_slots=2, max_k=15,
        )
        pipe = RetrievalPipeline(
            retriever=PassthroughRetriever(),
            session_manager=SessionStateManager(config),
            logger=logger,
            config=config,
            tool_registry=registry,
        )
        await pipe.get_tools_for_list("session-1")
        # May or may not have exploration tools depending on implementation detail
        # At minimum, should not crash
        assert isinstance(logger.events[0].exploration_tools, list)

    @pytest.mark.asyncio
    async def test_exploration_not_in_shadow(self) -> None:
        registry = _build_registry(25)
        logger = _CapturingLogger()
        config = RetrievalConfig(
            enabled=True, rollout_stage="shadow",
            enable_exploration=True, exploration_slots=2,
        )
        pipe = RetrievalPipeline(
            retriever=PassthroughRetriever(),
            session_manager=SessionStateManager(config),
            logger=logger,
            config=config,
            tool_registry=registry,
        )
        await pipe.get_tools_for_list("session-1")
        # Shadow sessions should NOT get exploration injection
        assert logger.events[0].exploration_tools == []

    def test_ranking_event_exploration_default(self) -> None:
        ev = RankingEvent(session_id="s", turn_number=0, catalog_version="v1")
        assert ev.exploration_tools == []
